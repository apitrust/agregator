```
kubectl apply -n keycloak -f - <<EOF
apiVersion: apisix.apache.org/v2
kind: ApisixRoute
metadata:
  name: keycloak1-route
spec:
  http:
  - name: keycloak1
    match:
      paths:
      - /
      - /admin/*
      - /resources/*
      - /realms/*
      - /welcome-content/*
    backends:
       - serviceName: auth-keycloak
         servicePort: 80
EOF

```
